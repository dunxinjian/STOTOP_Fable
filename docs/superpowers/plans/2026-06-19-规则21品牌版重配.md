# -*- coding: utf-8 -*-
import json, sys, subprocess, os
ROOT = r'E:\STOTOP_Fable'
# 1. code->fid (账套2) via dbq
def dbq(sql, out):
    subprocess.run(['python', os.path.join(ROOT,'Temp','dbq.py'), sql, '--out', out], check=True)
dbq("SELECT FID,FCode FROM FIN科目 WHERE FAccountSetId=2", os.path.join(ROOT,'Temp','_c2f.txt'))
rs = json.load(open(os.path.join(ROOT,'Temp','_c2f.txt'), encoding='utf-8'))[0]
code2fid = {str(r[1]): r[0] for r in rs['rows']}
# express_brand 申通 / business_unit 申通经营单元 fid: 从 Step1 _aux.txt 读取后填入(若无 business_unit 则设 None)
EXPRESS_BRAND_FID = None  # ← 填 Step1 查到的 express_brand=申通 fid
EXPRESS_BRAND_CODE = None # ← 填 code
BUSINESS_UNIT_FID = None  # ← 填申通经营单元 fid;无则保持 None(该维不配)

v6 = json.load(open(os.path.join(ROOT,'Temp','rule21_mapping_v6.json'), encoding='utf-8'))
groups = []
order = 1
for c in v6['combos']:
    code = str(c['target']['code'])
    fid = code2fid.get(code)
    assert fid, f"code {code} 未在账套2找到 fid"
    pay = c['payType']
    amount_field = 'F发生额收入' if pay == '总部应付' else 'F发生额支出'
    credit_dir = c['acctDirection']  # 贷(应付收入) / 借(应收成本)
    debit_dir = '借' if credit_dir == '贷' else '贷'
    # 损益行辅助核算四维
    aux = [{"auxType":"outlet","sourceType":"dynamic","sourceField":"F网点编号","matchBy":"code"}]
    if EXPRESS_BRAND_FID:
        aux.append({"auxType":"express_brand","sourceType":"fixed","fixedItemId":EXPRESS_BRAND_FID,"fixedItemCode":EXPRESS_BRAND_CODE})
    bd = c.get('businessDirection')
    if bd: aux.append({"auxType":"business_direction","sourceType":"fixed","fixedValue":bd})
    if BUSINESS_UNIT_FID:
        aux.append({"auxType":"business_unit","sourceType":"fixed","fixedItemId":BUSINESS_UNIT_FID})
    pnl_line = {"lineNo":1,"direction":credit_dir,"accountId":fid,"accountCode":code,
                "amountField":amount_field,"summaryTemplate":"{F业务摘要}","status":1,"displayOrder":1,
                "auxiliaryConfigs":aux}
    borrow_aux = []
    if EXPRESS_BRAND_FID:
        borrow_aux.append({"auxType":"express_brand","sourceType":"fixed","fixedItemId":EXPRESS_BRAND_FID,"fixedItemCode":EXPRESS_BRAND_CODE})
    borrow_fid = code2fid.get("220201"); assert borrow_fid == 700125, f"220201 fid漂移:{borrow_fid}"
    borrow_line = {"lineNo":2,"direction":debit_dir,"accountId":borrow_fid,"accountCode":"220201",
                   "amountField":amount_field,"summaryTemplate":"{F业务摘要}","status":1,"displayOrder":2,
                   "auxiliaryConfigs":borrow_aux}
    groups.append({"id":"g%03d"%order,"name":c['feeName'],"order":order,
                   "exactCodes":[c['feeCode']],"exactCategories":None,"categoryKeywords":None,
                   "summaryKeywords":None,"fallthrough":False,"amountAggregation":"ROW",
                   "lines":[pnl_line, borrow_line]})
    order += 1
config = {"mode":"rulesBased","version":2,"voucherWord":"记","dateField":"F业务日期",
          "stagingTable":"STG申通总部交易明细","accountSetId":2,
          "keyFields":["F科目编码","F费用名称","F业务摘要"],"unmatchedAction":"createDraft",
          "filterConditions":[],
          "matchingLayers":{"exactMatchField":"F科目编码","categoryField":"F费用名称","summaryField":"F业务摘要"},
          "ruleGroups":groups}
json.dump(config, open(os.path.join(ROOT,'Temp','rule21_brand_config.json'),'w',encoding='utf-8'), ensure_ascii=False)
print("groups:", len(groups))
```

    internal static CreateVoucherRequest BuildDraftRequest(
        RulesBasedVoucherConfigV2 config, List<IDictionary<string, object>> rows,
        DateTime voucherDate, long periodId, long batchId, string scopeId)
    {
        decimal total = 0m;
        foreach (var r in rows)
            foreach (var col in new[] { "F发生额收入", "F发生额支出" })
                if (r.TryGetValue(col, out var v) && decimal.TryParse(v?.ToString(), out var d)) total += Math.Abs(d);
        long acc = config.DraftPlaceholderAccountId ?? 0;
        var entries = new List<CreateVoucherEntryRequest> {
            new() { LineNo = 1, Summary = "[待补录]未匹配行", AccountId = acc, DebitAmount = total, CreditAmount = 0 },
            new() { LineNo = 2, Summary = "[待补录]未匹配行", AccountId = acc, DebitAmount = 0, CreditAmount = total },
        };
        return new CreateVoucherRequest {
            VoucherWord = config.VoucherWord, Date = voucherDate, PeriodId = periodId,
            Source = "费用支出", Remark = $"[待补录] 数据导入未匹配(V2) - 批次{batchId}",
            DataScopeId = scopeId, Entries = entries
        };
    }
```
