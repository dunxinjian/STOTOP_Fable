```python
# -*- coding: utf-8 -*-
import re
t = open(r'E:/STOTOP_Fable/Temp/InsertBrandAccounts.cs.txt', encoding='utf-8').read()
cols_n = 16  # 去F组织ID后列数: FID..F启用期间 共16
bad = 0; n = 0
for m in re.finditer(r'INSERT INTO \[FIN科目\] \(([^)]*)\) VALUES \((.*?)\);', t):
    n += 1
    ccount = m.group(1).count(',') + 1
    # 按逗号切值(忽略 N'…' 内逗号)
    raw = m.group(2); vals = []; cur = ''; inq = False
    for ch in raw:
        if ch == "'": inq = not inq; cur += ch
        elif ch == ',' and not inq: vals.append(cur); cur = ''
        else: cur += ch
    vals.append(cur)
    if ccount != cols_n or len(vals) != cols_n:
        bad += 1
        if bad <= 5: print('MISMATCH', m.group(2)[:80], 'cols=', ccount, 'vals=', len(vals))
print(f'共 {n} 条 INSERT; 列数!=值数 的有 {bad} 条; 期望列数={cols_n}')
```
